    def D(p=None, E=None, m=None, K1=0., K2=0., signs=None, order=(0, 1, 2, 3)):

    i, j, k, l = order
    sksl = signs[k] * signs[l]
    sisjsksl = signs[i] * signs[j] * signs[k] * signs[l]

    result = 0.

    if K1 != 0:
        result += K1 * (E[0]*E[1]*E[2]*E[3] * D1(*p) + sisjsksl * D3(*p))
