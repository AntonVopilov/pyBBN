        def totals(self):

        particle_totals = parmap(lambda particle: (
            particle.energy_density(),
            particle.numerator(),
            particle.denominator()
        ), self.particles)

        rho = 0
        numerator = 0
        denominator = 0

        for particle in particle_totals:
            rho += particle[0]
            numerator += particle[1]
            denominator += particle[2]

        return rho, numerator, denominator
