    #

    For intermediate regime equilibrium particles, density, energy density and pressure are obtained through integration of distribution function

    from scipy import integrate
import numpy
from common import GRID, PARAMS


name = 'intermediate'
    #

    Particle density

    \begin{equation} g \int \frac{p^2 dp}{2 \pi^2} f\left( \frac{p}{T} \right) \end{equation}

    def density(particle):

    if not particle._density:
        particle._density, _ = integrate.quad(
            lambda p: (
                particle.distribution_function(
                    particle.energy(p) / particle.T
                ) * p**2 * particle.dof / 2. / numpy.pi**2
            ), GRID.MIN_MOMENTUM, GRID.MAX_MOMENTUM
        )
    return particle._density

    Particle energy density

    \begin{equation}
       I_\rho = \frac{g}{2 \pi^2} p^2 E(p) \, f\left( \frac{E(p)}{T} \right)
   \end{equation}
    
        
    
        
    
          
    def energy_density_integrand(p, particle):

    E = particle.energy(p)
    return (
        particle.distribution_function(E / particle.T)
        * p**2 * E
        * particle.dof / 2. / numpy.pi**2
    )
    
        
    
        
    
    
    
  
    
  
    
    
    
    
        
    
          
    
            #
          
    
          
    \begin{equation}
       \rho = \int dp I_\rho
   \end{equation}
    
        
    
        
    
          
    def energy_density(particle):

    if not particle._energy_density:
        particle._energy_density, _ = integrate.quad(
            lambda p: energy_density_integrand(p, particle),
            GRID.MIN_MOMENTUM, GRID.MAX_MOMENTUM
        )
    return particle._energy_density

     Particle pressure 
    
        
    
        
    
          
    
        
    
        
    
    
    
  
    
  
    
    
    
    
        
    
          
    
            #
          
    
          
    \begin{equation}
       I_P = \frac{g}{6 \pi^2} \frac{p^4}{E(p)} \, f\left(\frac{E(p)}{T}\right)
   \end{equation}
    
        
    
        
    
          
    def pressure_integrand(p, particle):

    E = particle.energy(p)
    return (
        particle.distribution_function(E / particle.T)
        * p**4 / E
        * particle.dof / 6. / numpy.pi**2
    )
    
        
    
        
    
    
    
  
    
  
    
    
    
    
        
    
          
    
            #
          
    
          
    \begin{equation}
       P = \int dp I_P
   \end{equation}
    
        
    
        
    
          
    def pressure(particle):

    if not particle._pressure:
        particle._pressure, _ = integrate.quad(
            lambda p: pressure_integrand(p, particle),
            GRID.MIN_MOMENTUM, GRID.MAX_MOMENTUM
        )
    return particle._pressure

    \begin{equation}
       \frac{M^2 x}{m^2 a T} * I(2)

    def numerator(particle):

    return particle.mass**2 * PARAMS.x / PARAMS.m**2 / PARAMS.aT * I(particle, 2)

    \begin{equation}
       \frac{(I(4) + M_N^2 I(2))}{(a T)^2}
   \end{equation}
    
        
    
        
    
          
    def denominator(particle):

    return (I(particle, 4) + particle.mass_normalized**2 * I(particle)) / PARAMS.aT**2
    
        
    
        
    
    
    
  
    
  
    
    
    
    
        
    
          
    
            #
          
    
          
    \begin{equation}
       I(n) = \frac{g}{2 \pi^2} \int y^n dy \frac{ e^{-\frac{E_N(y)}{a T}} } { \left(e^{-\frac{E_N(y)}{a T}} + \eta \right)^2 }
   \end{equation}
    
        
    
        
    
          
    def I(particle, y_power=2):

    return particle.dof / 2 / numpy.pi**2 * integrate.quad(
        lambda y: (
            y**y_power * numpy.exp(-particle.energy_normalized(y) / PARAMS.aT)
            / (numpy.exp(-particle.energy_normalized(y) / PARAMS.aT) + particle.eta) ** 2
        ), GRID.MIN_MOMENTUM, GRID.MAX_MOMENTUM
